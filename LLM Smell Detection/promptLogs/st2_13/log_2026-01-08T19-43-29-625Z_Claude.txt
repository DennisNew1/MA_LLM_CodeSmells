     *                            - the parts before and after the '@' may contain one or several '.' as separators
     *                            - two '.' characters must not be directly next to each other (so "test@this..example.com" is invalid)
     *                            - the part after the '@' must end with ".de", ".at", ".ch", ".com", ".org"
     *                            (for simplicity we do not allow any other domains)

     * @throws ShopException if ...

        String[] vorATundNachAT = mailAddressAsString.split("@");

        if (vorATundNachAT.length != 2 || vorATundNachAT[0].isEmpty() || vorATundNachAT[1].isEmpty()) {
            throw new ShopException("the part before the '@' and the part after the '@' must not be empty");
        }

        if (mailAddressAsString.contains("..") || mailAddressAsString.contains(" ")) {
            throw new ShopException("two '.' characters must not be directly next to each other or contain any whitespace characters the vorATundNachAT before and after the '@'");
        }

        if (!mailAddressAsString.matches("[a-zA-Z0-9.@]+")) {

    public ResponseEntity<List<OrderDTO>> getAllOrders(@RequestParam(required = false) String mailAddress,@RequestParam(required = false) String filter) {
            List<OrderDTO> orderdtos = new ArrayList<>();

            List<Order> orders = orderService.findOrdersByMailAddressString(mailAddress);

            if (mailAddress == null || !orderService.existsByMailAddressString(mailAddress)) {
                return new ResponseEntity<>(orderdtos, HttpStatus.OK);
            }

            for (Order order : orders) {//price
                orderdtos.add(mapper.mapToDto(order));
            }
            List<OrderDTO> latestOrderDto = new ArrayList<>();

            if (filter != null && filter.equals("latest")) {
                latestOrderDto.add(orderdtos.get(orderdtos.size() - 1 ));
                return new ResponseEntity<>( latestOrderDto, HttpStatus.OK);
            }
            return new ResponseEntity<>(orderdtos, HttpStatus.OK);
        }
}

@Service
public class PaymentService implements PaymentUseCases {
    private final UserPaymentInterface userService;
    private final PaymentRepository paymentRepository;

    public PaymentService(
            UserPaymentInterface userService, PaymentRepository paymentRepository) {
        this.userService = userService;
        this.paymentRepository = paymentRepository;
    }

    /**
     * Authorizes a payment for a user (identified by his/her mail address) for a given amount
     *
     * @param userMailAddress
     * @param amount
     * @return the id of the payment, if successfully authorized
     * @throws ShopException if ...
     *                       - userMailAddress is null or empty
     *                       - the user with the given mail address does not exist
     *                       - the amount is <= 0.00 EUR
     *                       - the payment cannot be processed, because it is over the limit of 500.00 EUR
     */
    @Override
    public UUID authorizePayment(MailAddressType userMailAddress, MoneyType amount) {
        if (userMailAddress == null ) {
            throw new ShopException("userMailAddress is null or empty");
        }

        if (!userService.existsByMailAddress((MailAddress) userMailAddress)) {
            throw new ShopException("the user with the given mail address does not exist");
        }
        if (amount == null) {
            throw new ShopException("amount is null");
        }
        if (amount.getAmount() <= 0.00F || amount.getAmount() > 500.00F) {
            throw new ShopException("the payment cannot be processed, because it is over the limit of 500.00 EUR");
        }
        Payment newPayment = new Payment(amount);
        userService.addNewPayment((MailAddress) userMailAddress,newPayment);
        paymentRepository.save(newPayment);
        return newPayment.paymentID;
    }


    /**
     * Returns the total amount of payments (over the complete history) for a user
     * (identified by his/her mail address)
     *
     * @param userMailAddress
     * @return the total amount of payments for the user with the given mailAddress,
     * or 0.00 EUR if the user has not made any payments yet.
     * @throws ShopException if ...
     *                       - userMailAddress is null or empty
     *                       - the user with the given mailAddress does not exist
     */
    @Override
    public MoneyType getPaymentTotal(MailAddressType userMailAddress) {


        List<Payment> payments = userService.returnPaymentMap((MailAddress) userMailAddress);
        Float total = 0.00F;

        for (Payment payment : payments) {
            total += payment.getMoney().getAmount();
        }
        return new Money(total,"EUR");
    }

     * Deletes all payment history, for all users.

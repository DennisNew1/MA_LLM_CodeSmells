@Embeddable
@EqualsAndHashCode
@NoArgsConstructor (access = AccessLevel.PROTECTED)
public class Email implements EmailType {

    private String emailString;

    public String getEmailString() {
        return emailString;
    }

    public String toString() {
        return emailString;
    }

    public static Email of(String emailAsString) {
        if (emailAsString == null || emailAsString.isBlank()
           || emailAsString.chars().filter(atCharInString -> atCharInString == '@').count() != 1)
            throw new ShopException("Invalid email!");

        String[] validDomain = {".de", ".at", ".ch", ".com", ".org"};
        int domainCounter = 0;
        String[] splitEmail = emailAsString.split("@");

        if (splitEmail.length != 2)
            throw new ShopException("Invalid email!");

        String beforeAt = splitEmail[0];
        String afterAt = splitEmail[1];

        if (beforeAt.isBlank() || beforeAt.contains(" ")
           || beforeAt.contains("..")
           || afterAt.isBlank() || afterAt.contains(" ")
           || afterAt.contains(".."))
            throw new ShopException("Invalid email!");
        if (!beforeAt.matches("[a-zA-Z0-9.]+")
           || !afterAt.matches("[a-zA-Z0-9.]+"))
            throw new ShopException("Invalid email!");
        for (String domain : validDomain) {
            if (!afterAt.endsWith(domain)) {
                if (domainCounter == validDomain.length - 1)
                    throw new ShopException("Invalid domain!");
            } else break;
            domainCounter++;
        }
        return new Email(emailAsString);
    }
    private Email(String email) {
        this.emailString = email;
    }
}

    public Float getAmount() {
        return amount;
    }

    @Override
    public String getCurrency() {
        return currency;
    }

    @Override
    public MoneyType add(MoneyType otherMoney) {
        if (otherMoney == null || !otherMoney.getCurrency().equals(this.currency))
            throw new ShopException("Invalid addition!");
        return new Money(getAmount() + otherMoney.getAmount(), getCurrency());
    }

    @Override
    public MoneyType subtract(MoneyType otherMoney) {
        if (otherMoney == null
           || !otherMoney.getCurrency().equals(this.currency)
           || this.amount < otherMoney.getAmount())
            throw new ShopException("Invalid subtraction!");
        return new Money(getAmount() - otherMoney.getAmount(), getCurrency());
    }

    @Override
    public MoneyType multiplyBy(int factor) {
        if (factor < 0)
            throw new ShopException("Invalid factor!");
        return new Money(getAmount() * factor, getCurrency());
    }

    @Override
    public boolean largerThan(MoneyType otherMoney) {
        if (otherMoney == null || !this.currency.equals(otherMoney.getCurrency()))
            throw new ShopException("Invalid othermoney!");
        return this.amount > otherMoney.getAmount();
    }

    public static Money of( Float amount, String currency ) {
        if (amount == null || amount < 0.00F
           || currency == null
           || !currency.equals("EUR")
           && !currency.equals("CHF"))
            throw new ShopException("Invalid money!");
        return new Money(amount, currency);
    }

    private Money(Float amount, String currency) {
        this.amount = amount;
        this.currency = currency;
    }

        ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
        String path = packageName.replace( '.', '/' );
        Enumeration<URL> resources = classLoader.getResources( path );
        int count = 0;
        Class<?> result = null;
        while (resources.hasMoreElements()) {
            URL resource = resources.nextElement();
            File directory = new File( URLDecoder.decode( resource.getFile(), StandardCharsets.UTF_8 ) );
            File[] files = directory.listFiles();
            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );

                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;

    public <T> boolean nullOrEmpty (T thing) {
        if (thing == null)
            return true;
        if (thing instanceof String)
            return ((String) thing).isEmpty();
        if (thing instanceof Integer)
            return (Integer) thing == 0;
        if (thing instanceof Map)
            return ((Map<?, ?>) thing).isEmpty();
        return false;
    }

    public EmailType getEmail(Delivery delivery) {
        return delivery.getDeliveryRecipientEmail();
    }

    public int getDeliveryQuantity(Map<UUID, Integer> deliveryContent) {

            deliveryQuantity += quantity;
        }

        return deliveryQuantity;
    }

    @Override
    public UUID triggerDelivery(UserType deliveryRecipient, Map<UUID, Integer> deliveryContent) {
        if (deliveryRecipient == null
           || nullOrEmpty(deliveryRecipient.getName()))
            throw new ShopException("Not a valid delivery recipient!");
        if (nullOrEmpty(deliveryContent))
            throw new ShopException("Delivery content is not valid!");
        if (deliveryContent.size() > 20 || getDeliveryQuantity(deliveryContent) > 20)
            throw new ShopException("Too many products!");

        Delivery newDelivery = new Delivery();
        newDelivery.setDeliveryRecipientEmail((Email) deliveryRecipient.getEmail());

        for (UUID productId : deliveryContent.keySet()) {
            if (!productService.existsById(productId))
                throw new ShopException("No such product exists!");
            newDelivery.getDeliveryContent().add(
            orderPositionService.newDeliveryOrderPosition(getEmail(newDelivery),
            productService.findById(productId),
            deliveryContent.get(productId))
            );
        }
        deliveryRepository.save(newDelivery);
        return newDelivery.getDeliveryId();
    }

    @Override
    public Map<UUID, Integer> getDeliveryHistory(EmailType userEmail) {
        if (userService.checkNullOrEmptyOrExists(userEmail))
            throw new ShopException("No such user exists!");

        return orderPositionService.getMapOfDeliveriesForUser(userEmail);
    }

    public void deleteDeliveryHistory() {
        orderPositionService.deleteDeliveries();
        deliveryRepository.deleteAll();
    }

     * find all orders for user with emailID, ordered by Time
     * GET /orders?email=...
     * The returned Data needs to be an array of orders. If email is not found,
     * or has no orders, then an empty array is returned.
     * @returns Array<OrderDto>
     */
    @GetMapping(value = "/orders")
    public ResponseEntity<ArrayList<OrderDto>> getForSpecificEmail(
            @RequestParam(required = false) String email,
            @RequestParam(required = false) String filter) {
        ArrayList<OrderDto> orderDtos = new ArrayList<>();
        try {
            Email emailString = Email.of(email);
        } catch (ShopException shopException) {
            return new ResponseEntity<>(orderDtos, HttpStatus.OK);
        }
        Email emailString = Email.of(email);
        if (orderService.existsByUserEmail(emailString)) {
            if (filter != null && filter.equals("latest")) {
                List<Order> orders = new ArrayList<>(orderService.findByUserEmail(emailString));
                Order latestOrder = orders.stream().max(Comparator.comparing(Order::getSubmissionDate)).get();
                orderDtos.add(orderDtoMapper.mapToDto(latestOrder));
            } else {
                for (Order order : orderService.findByUserEmail(emailString)) {
                    orderDtos.add(orderDtoMapper.mapToDto(order));
                }
            }
        }
        return ResponseEntity.ok().body(orderDtos);
    }

package thkoeln.archilab.ecommerce.solution.order.application;

import thkoeln.archilab.ecommerce.solution.order.domain.entity.Order;
import thkoeln.archilab.ecommerce.usecases.domainprimitivetypes.MoneyType;

import java.util.ArrayList;
import java.util.List;

public class OrderDtoMapper {
    private ModelMapper modelMapper = new ModelMapper();
    private OrderPositionDtoMapper orderPositionDtoMapper = new OrderPositionDtoMapper();

    public OrderDto mapToDto(Order order) {
        OrderDto orderDto = new OrderDto();
        MoneyType moneyValue = order.getOrderPositions().get(0).getProduct().getSellPrice();
        moneyValue = moneyValue.subtract(moneyValue);

        for (AbstractOrderPosition orderPosition : order.getOrderPositions()) {
            int orderPositionQuantity = orderPosition.getQuantity();
            MoneyType orderPositionProductSellPrice = orderPosition.getProduct().getSellPrice();
            moneyValue = moneyValue.add(orderPositionProductSellPrice.multiplyBy(orderPositionQuantity));
        }


        orderDto.setId(order.getOrderId());
        orderDto.setEmail(order.getUserEmail());
        orderDto.setPrice((Money) moneyValue);
        List<OrderPositionDto> orderPositionDtos = new ArrayList<>();

        for (AbstractOrderPosition orderPosition : order.getOrderPositions()) {
            orderPositionDtos.add(orderPositionDtoMapper.orderPositionToOrderPositionDto(orderPosition));
        }

        orderDto.setOrderPositions(orderPositionDtos);

        return orderDto;
    }

    public Order mapToEntity (OrderDto orderDto) {
        return modelMapper.map(orderDto, Order.class);
    }
}

            Email userEmail,
            UUID productId,
            int quantity,
            String comment,
            URI returnURI)
    {
        ShoppingBasketDto shoppingBasketDto = new ShoppingBasketDto();

        if (quantity > 0) {
            try {
                shoppingBasketService.addProductToShoppingBasket(userEmail, productId, quantity);
            } catch (ShopException shopException) {
                return ResponseEntity.status(HttpStatus.CONFLICT).build();
            }
            if (comment != null)
                shoppingBasketService.setCommentForShoppingBasketPosition(userEmail, productId, comment);

            shoppingBasketDto = shoppingBasketDtoMapper.mapToDto(shoppingBasket);
            return ResponseEntity.created(returnURI).body(shoppingBasketDto);
        } else if (quantity < 0) {
            quantity = -quantity;

            if (comment != null)
                shoppingBasketService.setCommentForShoppingBasketPosition(userEmail, productId, comment);
            try {
                shoppingBasketService.removeProductFromShoppingBasket(userEmail, productId, quantity);
            } catch (ShopException shopException) {
                return ResponseEntity.status(HttpStatus.CONFLICT).build();
            }

            shoppingBasketDto = shoppingBasketDtoMapper.mapToDto(shoppingBasket);
            return ResponseEntity.created(returnURI).body(shoppingBasketDto);
        }

        return ResponseEntity.created(returnURI).body(shoppingBasketDto);

            URI returnURI = ServletUriComponentsBuilder

                    .toUri();

            if (quantity == null) {

            }

            if (shoppingBasketService.productInBasket(userEmail, productId)) {

                    shoppingBasketService.setCommentForShoppingBasketPosition(userEmail, productId, comment);
                }

                shoppingBasketDto = shoppingBasketDtoMapper.mapToDto(shoppingBasket);

            return ResponseEntity.created(returnURI).body(shoppingBasketDto);

    @Override
    public void addProductToShoppingBasket(EmailType userEmail, UUID productId, int quantity) {
        if (userEmail == null || productId == null) throw new ShopException("Invalid input!");
        if (userService.checkNullOrEmptyOrExists(userEmail)) throw new ShopException("User does not exist!");
        if (!productService.existsById(productId)) throw new ShopException("No such product exists!");
        if (!warehouseService.existsById(productId)) throw new ShopException("No such product in warehouse!");
        if (quantity < 0 || warehouseService.getQuantity(productId) < quantity)
            throw new ShopException("Invalid quantity!");

        if (getState(userEmail) != ShoppingBasketState.PAYMENT_AUTHORIZED) {
            if (getState(userEmail) == ShoppingBasketState.FILLED
                    && productInBasket(userEmail, productId)) {
                warehouseService.reduceQuantity(productId, quantity);
                shoppingBasketInternalService.addNewQuantityInOrderPosition(userEmail, productId, quantity);
            } else {
                warehouseService.reduceQuantity(productId, quantity);
                shoppingBasketInternalService.addToShoppingBasketPositions(
                        userEmail,
                        orderPositionService.newOrderPosition(userEmail, productService.findById(productId), quantity)
                );
                shoppingBasketInternalService.changeStateTo(userEmail, ShoppingBasketState.FILLED);
            }
        }
    }

    @Override
    public void removeProductFromShoppingBasket(EmailType userEmail, UUID productId, int quantity) {
        if (userService.checkNullOrEmptyOrExists(userEmail))
            throw new ShopException("User does not exist!");
        if (productService.checkNullOrEmpty(productId))
            throw new ShopException("No such product exists!");
        if (quantity < 0) throw new ShopException("Invalid quantity");

        if (!basketEmpty(userEmail) && productInBasket(userEmail, productId)) {
            shoppingBasketInternalService.reduceQuantityInShoppingBasketPosition(userEmail, productId, quantity);
            orderPositionService.deleteAllEmptyOrderPositions();
            warehouseService.addQuantity(productId, quantity);
        } else throw new ShopException("Order with that product does not exist");

        if (basketEmpty(userEmail))
            shoppingBasketInternalService.changeStateTo(userEmail, ShoppingBasketState.EMPTY);

    }

        if (productService.checkNullOrEmpty(productId) || !productService.existsById(productId))

    public MoneyType getShoppingBasketMoneyValue(EmailType userEmail) {
        MoneyType moneyValue = getShoppingBasketPositionsOfUserWithEmail(userEmail).get(0).getProduct().getSellPrice();
        moneyValue = moneyValue.subtract(moneyValue);

        for (AbstractOrderPosition orderPosition :
                shoppingBasketInternalService.getShoppingBasketPositions(userEmail)) {
            int orderPositionQuantity = shoppingBasketInternalService.getQuantity(orderPosition);
            MoneyType orderPositionProductSellPrice = getSellPrice(orderPosition);
            moneyValue = moneyValue.add(orderPositionProductSellPrice.multiplyBy(orderPositionQuantity));
        }

        return moneyValue;
    }

        for (ShoppingBasket shoppingBasket : shoppingBasketInternalService.findAll()) {
            if (productInBasket(shoppingBasket.getUserEmail(), productId)) {
                reservedQuantity += shoppingBasketInternalService.getQuantityOfProductInBasket(shoppingBasket.getUserEmail(), productId);
            }

    }

    @Override
    public void checkout(EmailType userEmail) {
        if (userEmail == null || userService.checkNullOrEmptyOrExists(userEmail))
            throw new ShopException("User does not exist!");
        if (!shoppingBasketInternalService.existsByUserEmail(userEmail))
            throw new ShopException("Cart is empty!");
        if (shoppingBasketInternalService.getShoppingBasketPositions(userEmail).isEmpty())
            throw new ShopException("Cart is empty!");

        if (paymentService.authorizePayment(userEmail, getShoppingBasketAsMoneyValue(userEmail)) != null)
            shoppingBasketInternalService.changeStateTo(userEmail, ShoppingBasketState.PAYMENT_AUTHORIZED);
        if (isPaymentAuthorized(userEmail)) {
            if (deliveryService.triggerDelivery(userService.findByEmail(userEmail), getShoppingBasketAsMap(userEmail)) != null) {
                shoppingBasketInternalService.changeStateTo(userEmail, ShoppingBasketState.DELIVERY_TRIGGERED);
            }
            if (isOrderDelivered(userEmail)) {
                clearShoppingBasketPositions(userEmail);
                List<AbstractOrderPosition> orderPositions = new ArrayList<>(orderPositionService.findAllOrdersForUser(userEmail));
                orderService.newOrder(userEmail, orderPositions);
                orderPositionService.markOrderPositionsOfUserAsHistory(userEmail);
                shoppingBasketInternalService.changeStateTo(userEmail, ShoppingBasketState.EMPTY);
            }
        }
    }

    }

        orderService.deleteAll();
        orderPositionService.deleteAll();
    }

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import thkoeln.archilab.ecommerce.solution.orderposition.application.OrderPositionService;
import thkoeln.archilab.ecommerce.solution.warehouse.domain.entity.Warehouse;
import thkoeln.archilab.ecommerce.usecases.InventoryManagementUseCases;
import thkoeln.archilab.ecommerce.ShopException;

import java.util.UUID;

@Service
public class InventoryManagementService implements InventoryManagementUseCases {

    private final ProductService productService;

    private final WarehouseService warehouseService;

    private final WarehouseShoppingBasketService shoppingBasketService;

    private final OrderPositionService orderPositionService;

    @Autowired
    public InventoryManagementService(
            ProductService productService,
            WarehouseService warehouseService,
            WarehouseShoppingBasketService shoppingBasketService,
            OrderPositionService orderPositionService
    ) {
        this.productService = productService;
        this.warehouseService = warehouseService;
        this.shoppingBasketService = shoppingBasketService;
        this.orderPositionService = orderPositionService;
    }

    @Override
    public void addToInventory(UUID productId, int addedQuantity) {
        if (productId == null || !productService.existsById(productId))
            throw new ShopException("No such product exists");
        if (addedQuantity <= 0)
            throw new ShopException("The Quantity has to be at least 0!");

        warehouseService.addToInventory(productId, addedQuantity);
    }

    public void removeSomeReservedFromCart(UUID productId, int newTotalAndReservedDifference){
        if (newTotalAndReservedDifference > 0) {
            orderPositionService.removeFromReserved(productId, newTotalAndReservedDifference);
            shoppingBasketService.removeFromReserved(productId);
            orderPositionService.deleteAllEmptyOrderPositions();
        }
    }

    @Override
    public void removeFromInventory(UUID productId, int removedQuantity) {
        if (warehouseService.checkNullOrEmpty(productId)
            || !warehouseService.existsById(productId)) throw new ShopException("No such product exists!");

        int reservedQuantity = shoppingBasketService.getReservedInventoryInShoppingBaskets(productId);
        int currentQuantity = warehouseService.findById(productId).getQuantity();
        int totalQuantity = reservedQuantity + currentQuantity;
        int newTotalQuantity = totalQuantity - removedQuantity;
        int newTotalAndReservedDifference = reservedQuantity - newTotalQuantity;

        if (currentQuantity + reservedQuantity < removedQuantity || removedQuantity <= 0)
            throw new ShopException("Invalid quantity amount!");

        removeSomeReservedFromCart(productId, newTotalAndReservedDifference);
        warehouseService.reduceQuantity(productId, removedQuantity);
    }

    @Override
    public void changeInventoryTo(UUID productId, int newTotalQuantity) {
        if (warehouseService.checkNullOrEmpty(productId)
            || !warehouseService.existsById(productId)) throw new ShopException("No such product exists");
        if (newTotalQuantity < 0) throw new ShopException("Invalid quantity amount!");

        int reservedQuantity = shoppingBasketService.getReservedInventoryInShoppingBaskets(productId);
        int newTotalAndReservedDifference = reservedQuantity - newTotalQuantity;

        removeSomeReservedFromCart(productId, newTotalAndReservedDifference);

        warehouseService.setNewQuantity(productId, newTotalQuantity);
    }

    @Override
    public int getAvailableInventory(UUID productId) {
        if (warehouseService.checkNullOrEmpty(productId)
            || !warehouseService.existsById(productId))
            throw new ShopException("No such product exists!");

        Warehouse product = warehouseService.findById(productId);
        int currentQuantity = product.getQuantity();
        int reservedQuantity = shoppingBasketService.getReservedInventoryInShoppingBaskets(productId);

        return currentQuantity + reservedQuantity;
    }
}

        save(findById(productId));

    }

        MetadataReaderFactory readerFactory = new CachingMetadataReaderFactory( resolver );

@EqualsAndHashCode
public class Money {

  private static final String[] ALLOWED_CURRENCIES = {"EUR", "CHF"};

  private Float amount;
  private String currency;

  /**
   * Unfortunately, Java interfaces cannot contain static methods. However, we expect the
   * implementing class to provide a static factory method (simply named "of(...)"),
   * which creates a money object from an amount and a currency (as string).
   * We specify this factory method here as a comment, using the Javadoc documentation style.
   *
   * @param amount the amount of money (must be >= 0)
   * @param currency the currency of the money (allowed values: "EUR", "CHF")
   * @return a new Money object with the given amount and currency
   * @throws ShopException if ...
   *   - amount is null
   *   - amount < 0
   *   - currency is null
   *   - currency is not one of the allowed values
   */
  public static Money of(final Float amount, final String currency) {
    return new Money(amount, currency);
  }

  private Money(final Float amount, final String currency) {
    setAmount(amount);
    setCurrency(currency);
  }

  private void setAmount(final Float amount) {
    if (amount == null)
      throw new ShopException("Amount must not be null");
    if (amount < 0)
      throw new ShopException("Amount must not be negative");

    this.amount = amount;
  }

  private void setCurrency(final String currency) {
    if (currency == null)
      throw new ShopException("Currency must not be null");
    if (Arrays.stream(ALLOWED_CURRENCIES).noneMatch(currency::equals))
      throw new ShopException("Currency must not be " + currency);

    this.currency = currency;
  }

  /**
   * @param otherMoney
   * @return this + otherMoney, as a new object
   * @throws ShopException if ...
   *                       - otherMoney is null
   *                       - otherMoney.currency != this.currency
   */
  public Money add(final Money otherMoney) {
    if (otherMoney == null)
      throw new ShopException("Other money must not be null");
    if (!otherMoney.getCurrency().equals(this.currency))
      throw new ShopException("Other money must not have a different currency");

    return Money.of(this.amount + otherMoney.getAmount(), this.currency);
  }

  /**
   * @param otherMoney
   * @return this - otherMoney, as a new object
   * @throws ShopException if ...
   *                       - otherMoney is null
   *                       - otherMoney.currency != this.currency
   *                       - otherMoney > this
   */
  public Money subtract(final Money otherMoney) {
    if (otherMoney == null)
      throw new ShopException("Other money must not be null");
    if (!otherMoney.getCurrency().equals(this.currency))
      throw new ShopException("Other money must not have a different currency");
    if (otherMoney.getAmount() > this.amount)
      throw new ShopException("Other money must not be larger than this");

    return Money.of(this.amount - otherMoney.getAmount(), this.currency);
  }

  /**
   * @param factor
   * @return this * factor, as a new object
   * @throws ShopException if ...
   *                       - factor < 0
   */
  public Money multiplyBy(final int factor) {
    if (factor < 0)
      throw new ShopException("Factor must not be negative");

    return Money.of(this.amount * factor, this.currency);
  }

  /**
   * @param otherMoney
   * @return true, if this > otherMoney
   * @throws ShopException if ...

  public boolean largerThan(final Money otherMoney) {

    if (!otherMoney.getCurrency().equals(this.currency))
      throw new ShopException("Other money must not have a different currency");

            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );
                    Class<?> clazz = Class.forName( className );
                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;
    }
}

  private static final int MAX_DELIVERY_PARTS = 20;

  private final DeliveryRepository deliveryRepository;
  private final CustomerService customerService;
  private final ItemCatalogService itemCatalogService;

  @Autowired
  public DeliveryService(DeliveryRepository deliveryRepository, @Lazy CustomerService customerService, @Lazy ItemCatalogService itemCatalogService) {
    this.deliveryRepository = deliveryRepository;
    this.customerService = customerService;
    this.itemCatalogService = itemCatalogService;
  }

  public UUID triggerDelivery(Customer deliveryRecipient, Map<UUID, Integer> deliveryContent) {
    if (deliveryRecipient == null)
      throw new ShopException("Delivery recipient is null");

    if (deliveryRecipient.getName() == null || deliveryRecipient.getName().isEmpty() ||
        deliveryRecipient.getEmail() == null ||
        deliveryRecipient.getAddress() == null)
      throw new ShopException("Delivery recipient is invalid");

    if (deliveryContent == null || deliveryContent.isEmpty())
      throw new ShopException("Delivery content is null or empty");

    if (deliveryContent.values().stream().reduce(0, Integer::sum) > MAX_DELIVERY_PARTS)
      throw new ConflictException("The total number of items in the delivery is > 20");

    deliveryContent.keySet().forEach(itemId -> {
      if (!itemCatalogService.isItemInCatalog(itemId))
        throw new ShopException("Item with the given id does not exist" + itemId);
    });
    var customer = customerService.getCustomer(deliveryRecipient.getEmail())
        .orElseThrow(() -> new ShopException("Customer with the given email does not exist"));

    var delivery = new Delivery();
    delivery.setCustomer(customer);
    delivery.setParts(
        deliveryContent
            .entrySet()
            .stream()
            .map(entry -> Map.entry(itemCatalogService.getItem(entry.getKey()), entry.getValue()))
            .map(DeliveryPart::new)
            .collect(Collectors.toList())
    );
    deliveryRepository.save(delivery);
    return delivery.getId();
  }

  public Map<UUID, Integer> getDeliveryHistory(Email customerEmail) {
    if (customerEmail == null)
      throw new ShopException("Customer email is null");

    if (!customerService.isCustomerRegistered(customerEmail))
      throw new ShopException("Customer with the given email does not exist");

    return deliveryRepository.findByCustomerEmail(customerEmail)
        .stream()
        .map(Delivery::getParts)
        .flatMap(Collection::stream)
        .map(orderPart -> Map.entry(orderPart.getItem().getId(), orderPart.getQuantity()))
        .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue, Integer::sum));
  }

  public void deleteDeliveryHistory() {
    deliveryRepository.deleteAll();
  }
}

        .orElseThrow(() -> new ShopException("Customer with the given email does not exist"));
    var payment = paymentRepository.findById(paymentId)
        .orElseThrow(() -> new ShopException("Payment with the given id does not exist"));
    var delivery = deliveryRepository.findById(deliveryId)
        .orElseThrow(() -> new ShopException("Delivery with the given id does not exist"));
    var order = new Order();
    order.setCustomer(customer);
    order.setPayment(payment);
    order.setDelivery(delivery);
    order.setParts(

            .entrySet()
            .stream()
            .map(entry -> Map.entry(itemCatalogService.getItem(entry.getKey()), entry.getValue()))
            .map(OrderPart::new)
            .collect(Collectors.toList())
    );

  }

    if (!customerService.isCustomerRegistered(customerEmail))

    return orderRepository.findByCustomerEmail(customerEmail)
        .stream()
        .map(Order::getParts)
        .flatMap(Collection::stream)
        .map(orderPart -> Map.entry(orderPart.getItem().getId(), orderPart.getQuantity()))
        .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue, Integer::sum));

  }

    var shoppingCartPart = new ShoppingCartPart(item, quantity, comment);

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.UUID;
import java.util.stream.Collectors;

@Entity
@NoArgsConstructor
@Getter
@Setter
public class ShoppingCart {

  @Id
  @GeneratedValue
  @Column(name = "shoppingcart_id")
  private UUID id;

  @OneToOne
  private Customer customer;

  @OneToOne
  private Payment payment;

  @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true)
  private List<ShoppingCartPart> parts = new ArrayList<>();

  public void addItem(ShoppingCartPart shoppingCartPart) {
    parts.stream()
        .filter(part -> part.getItem().getId().equals(shoppingCartPart.getItem().getId()))
        .findFirst()
        .ifPresentOrElse(
            part -> {
              part.increaseQuantity(shoppingCartPart.getQuantity());
              part.setComment(shoppingCartPart.getComment());
            },
            () -> parts.add(shoppingCartPart));
    setPayment(null);
  }

  public void removeItem(ShoppingCartPart shoppingCartPart) {
    parts.stream()
        .filter(part -> part.getItem().getId().equals(shoppingCartPart.getItem().getId()))
        .findFirst()
        .ifPresentOrElse(
            part -> {
              if (part.getQuantity() > shoppingCartPart.getQuantity()) {
                part.decreaseQuantity(shoppingCartPart.getQuantity());
                part.setComment(shoppingCartPart.getComment());
              } else if (part.getQuantity() == shoppingCartPart.getQuantity()) {
                parts.remove(part);
              } else {
                throw new ConflictException("Quantity of item in shopping cart is less than requested");
              }
            },
            () -> {
              throw new ShopException("Item not found in shopping cart");
            }
        );
    setPayment(null);
  }

  public Map<UUID, Integer> getShoppingCartAsMap() {
    return parts
        .stream()
        .collect(Collectors.toMap(part -> part.getItem().getId(), ShoppingCartPart::getQuantity));
  }

  public Map<UUID, Pair<Integer, String>> getCommentsAsMap() {
    return parts
        .stream()
        .collect(Collectors.toMap(part -> part.getItem().getId(), part -> Pair.of(part.getQuantity(), part.getComment())));
  }

  public Money getShoppingCartAsMoneyValue() {
    return parts
        .stream()
        .map(part -> part.getItem().getSellPrice().multiplyBy(part.getQuantity()))
        .reduce(Money::add)
        .orElseThrow();
  }

  public boolean isEmpty() {
    return parts.isEmpty();
  }
}

    private static Class<?> findImplementingClass( String packageName, Class<?> interfaceClass )
            throws ClassNotFoundException, IOException {
        ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
        String path = packageName.replace( '.', '/' );
        Enumeration<URL> resources = classLoader.getResources( path );
        int count = 0;
        Class<?> result = null;
        while (resources.hasMoreElements()) {
            URL resource = resources.nextElement();
            File directory = new File( URLDecoder.decode( resource.getFile(), StandardCharsets.UTF_8 ) );
            File[] files = directory.listFiles();
            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );
                    Class<?> clazz = Class.forName( className );
                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;
    }

        MetadataReaderFactory readerFactory = new CachingMetadataReaderFactory( resolver );

        Set<String> foundClasses = new HashSet<>();

        Arrays.stream( resolver.getResources( packageSearchPath ) ).forEach( resource -> {

        } );

    @Test
    public void testImpactOfInventoryCorrectionToSeveralShoppingCarts() {
        // given
        UUID itemId2 = (UUID) ITEM_DATA[2][0];
        EmailType customerEmail3 = CUSTOMER_EMAIL[3];
        EmailType customerEmail6 = CUSTOMER_EMAIL[6];
        EmailType customerEmail9 = CUSTOMER_EMAIL[9];

        // when
        shoppingCartUseCases.addItemToShoppingCart( customerEmail3, itemId2, 3 );
        shoppingCartUseCases.addItemToShoppingCart( customerEmail6, itemId2, 6 );
        shoppingCartUseCases.addItemToShoppingCart( customerEmail9, itemId2, 9 );
        inventoryManagementUseCases.removeFromInventory( itemId2, 2 );
        Map<UUID, Integer> cart31 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail3 );
        Map<UUID, Integer> cart61 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail6 );
        Map<UUID, Integer> cart91 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail9 );
        int reservedInventory21 = shoppingCartUseCases.getReservedInventoryInShoppingCarts( itemId2 );

        inventoryManagementUseCases.removeFromInventory( itemId2, 8 );
        Map<UUID, Integer> cart32 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail3 );
        Map<UUID, Integer> cart62 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail6 );
        Map<UUID, Integer> cart92 = shoppingCartUseCases.getShoppingCartAsMap( customerEmail9 );
        int quantity32 = cart32.get( itemId2 ) == null ? 0 : cart32.get( itemId2 );
        int quantity62 = cart62.get( itemId2 ) == null ? 0 : cart62.get( itemId2 );
        int quantity92 = cart92.get( itemId2 ) == null ? 0 : cart92.get( itemId2 );
        int reservedInventory22 = shoppingCartUseCases.getReservedInventoryInShoppingCarts( itemId2 );

        // then
        assertEquals( 3, cart31.get( itemId2 ) );
        assertEquals( 6, cart61.get( itemId2 ) );
        assertEquals( 9, cart91.get( itemId2 ) );
        assertEquals( 18, reservedInventory21 );
        assertEquals( 10, reservedInventory22 );
        assertEquals( reservedInventory22, quantity32 + quantity62 + quantity92 );
    }

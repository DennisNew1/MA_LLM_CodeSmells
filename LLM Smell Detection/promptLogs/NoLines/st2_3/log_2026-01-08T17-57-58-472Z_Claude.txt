@EqualsAndHashCode
public class Money implements MoneyType {

    private Float amount;
    private String currency;

    private Money(Float amount, String currency) {
        this.amount = amount;
        this.currency = currency;
    }

    protected Money() {
        //TODO: Delete
    }

    public static MoneyType of(Float amount, String currency ){
        if (amount == null ||currency == null) throw new ShopException("Empty values");
        if (amount < 0 ) throw new ShopException("Negative amount");
        if (!(currency.equals("EUR") || currency.equals("CHF"))) throw new ShopException("Not EUR or CHF");
        return new Money(amount, currency);
    }


    @Override
    public Float getAmount() {
        return amount;
    }

    @Override
    public String getCurrency() {
        return currency;
    }


    @Override
    public MoneyType add(MoneyType otherMoney) {
        if (otherMoney == null) throw new ShopException("Money == null");
        if(!this.currency.equals(otherMoney.getCurrency())) throw new ShopException("Currency not equal");
        return new Money(amount+ otherMoney.getAmount(), currency);
    }

    @Override
    public MoneyType subtract(MoneyType otherMoney) {
        if (otherMoney == null) throw new ShopException("Money == null");
        if(!this.currency.equals(otherMoney.getCurrency())) throw new ShopException("Currency not equal");
        if(this.amount < otherMoney.getAmount()) throw new ShopException("amount < otherMoney");

        return new Money(this.amount-otherMoney.getAmount(), this.currency);
    }

    @Override
    public MoneyType multiplyBy(int factor) {
        if (factor < 0) throw new ShopException("factor < 0");
        return new Money(this.amount*factor, this.currency);
    }

    @Override
    public boolean largerThan(MoneyType otherMoney) {
        if (otherMoney == null) throw new ShopException("Money == null");
        if(!this.currency.equals(otherMoney.getCurrency())) throw new ShopException("Currency not equal");

        return this.amount > otherMoney.getAmount();
    }


}

    private static Class<?> findImplementingClass( String packageName, Class<?> interfaceClass )
            throws ClassNotFoundException, IOException {
        ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
        String path = packageName.replace( '.', '/' );
        Enumeration<URL> resources = classLoader.getResources( path );
        int count = 0;
        Class<?> result = null;
        while (resources.hasMoreElements()) {
            URL resource = resources.nextElement();
            File directory = new File( URLDecoder.decode( resource.getFile(), StandardCharsets.UTF_8 ) );
            File[] files = directory.listFiles();
            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );
                    Class<?> clazz = Class.forName( className );
                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;
    }

    public UUID triggerDelivery(CustomerType deliveryRecipient, Map<UUID, Integer> deliveryContent) {
        if (deliveryRecipient == null || deliveryContent.size() == 0)
            throw new ShopException("deliveryContent is null or empty");
        int sum = 0;
        for (int quantity : deliveryContent.values()) sum += quantity;
        if (sum > 20) throw new ShopException("deliveryContent is > 20");

        Customer customer = customerService.findCustomerByEmail(deliveryRecipient.getEmail());

        Delivery newDelivery = new Delivery(customer);
        entityManger.persist(newDelivery);
        for (Map.Entry<UUID, Integer> entry : deliveryContent.entrySet()) {
            Good good = goodService.findGoodByGoodId(entry.getKey());
            if (good == null) throw new ShopException("Good not available");
            DeliveryPosition deliveryPosition = new DeliveryPosition(entry.getValue(), newDelivery, good);
            entityManger.persist(deliveryPosition);
        }

        shoppingBasketService.getShoppingBasketByCostumer(customer).setShoppingBasketState(ShoppingBasketState.DELIVERY_TRIGGERED);
        entityManger.persist(customer);

        return newDelivery.getId();
    }

    public Map<UUID, Integer> getDeliveryHistory(EmailType customerEmail) {
        if(customerEmail == null) throw new ShopException("Email is empty");
        Customer customer = customerService.findCustomerByEmail(customerEmail);
        if(customer == null) throw new ShopException("Customer doesn't exists");

        Map<UUID, Integer> deliveryHistory = new HashMap<>();

        List<Delivery> deliveries = deliveryRepository.findAllByCustomer(customer);
        for (Delivery delivery : deliveries) {
            List<DeliveryPosition> deliveryPositions = deliveryPositionRepository.findAllByDelivery(delivery);
            for (DeliveryPosition deliveryPosition : deliveryPositions) {
                int quantity = deliveryHistory.getOrDefault(deliveryPosition.getGood().getGoodId(), 0);
                deliveryHistory.put(deliveryPosition.getGood().getGoodId(), quantity + deliveryPosition.getQuantity());
            }
        }
        return deliveryHistory;
    }

    public void deleteDeliveryHistory() {
        deliveryPositionRepository.deleteAll();
        deliveryRepository.deleteAll();
    }

        if(purchasePrice.largerThan(salesPrice)) throw new ShopException("the sales price is lower than the purchase price");

    }

        if(orderService.getOrderPositionSizeByGood(good) != 0 || shoppingBasketService.getReservedPositionSizeByGood(good) != 0)
            throw new ShopException("the good is still reserved in a shopping basket, or part of a completed order");

            }else{
                totalPrice = totalPrice.add(orderPosition.getGood().getSalesPrice().multiplyBy(orderPosition.getQuantity()));
            }
        }

    public void addGoodToShoppingBasket(EmailType customerEmail, UUID goodId, int quantity) {
        Customer customer = customerService.findCustomerByEmail(customerEmail);
        if (customer == null) throw new ShopException("Customer not found!");
        Good good = goodService.findGoodByGoodId(goodId);
        if (good == null) throw new ShopException("the good id does not exist");
        if (quantity < 0) throw new ShopException("the quantity is negative");

        List<ReservedGood> reservedGoods = reservedGoodRepository.findAllByGood(good);
        int stock = good.getQuantity();
        for (ReservedGood reservedGood : reservedGoods) stock -= reservedGood.getQuantity();
        if (stock < quantity) throw new ShopException("the good is not available in the requested quantity");

        ShoppingBasket shoppingBasket = getShoppingBasketByCostumer(customer);

        ReservedGood reservedGood = reservedGoodRepository.findByGoodAndShoppingBasket(good, shoppingBasket);
        if(reservedGood == null) {
            reservedGood = new ReservedGood(quantity, good, shoppingBasket, null);
        }else{
            reservedGood.addQuantity(quantity);
        }
        entityManger.persist(reservedGood);

        //State
        shoppingBasket.setShoppingBasketState(ShoppingBasketState.FILLED);
        entityManger.persist(customer);
    }

    @Override
    public void removeGoodFromShoppingBasket(EmailType customerEmail, UUID goodId, int quantity) {
        Customer customer = customerService.findCustomerByEmail(customerEmail);
        if (customer == null) throw new ShopException("Customer not found");
        Good good = goodService.findGoodByGoodId(goodId);
        if (good == null) throw new ShopException("the good id does not exist");
        if (FunctionHelper.isNegativeOrNull(quantity)) throw new ShopException("the quantity is negative");
        ShoppingBasket shoppingBasket = getShoppingBasketByCostumer(customer);
        ReservedGood reservedGood = reservedGoodRepository.findByGoodAndShoppingBasket(good, shoppingBasket);
        if (reservedGood == null || reservedGood.getQuantity() < quantity)
            throw new ShopException("the good is not in the cart in the requested quantity");

        int newQuantity = reservedGood.removeQuantity(quantity);
        if(newQuantity == 0){
            reservedGoodRepository.delete(reservedGood);
        }else{
            entityManger.persist(reservedGood);
        }
        //State
        List<ReservedGood> reservedGoods = reservedGoodRepository.findAllByShoppingBasket(shoppingBasket);
        if(reservedGoods.size()==0){
            shoppingBasket.setShoppingBasketState(ShoppingBasketState.EMPTY);
        }else{
            shoppingBasket.setShoppingBasketState(ShoppingBasketState.FILLED);
        }
        entityManger.persist(customer);

            "Order",

    @Test
    public void testRequiredClassesDeclared() throws IOException {

        String packageSearchPath = "classpath*:" + basePackage.replace( '.', '/' ) + "/**/*.class";

        PathMatchingResourcePatternResolver resolver = new PathMatchingResourcePatternResolver();
        MetadataReaderFactory readerFactory = new CachingMetadataReaderFactory( resolver );

        Set<String> foundClasses = new HashSet<>();

        Arrays.stream( resolver.getResources( packageSearchPath ) ).forEach( resource -> {
            try {

                    className.startsWith(basePackage) && className.endsWith("." + requiredClass)

        });

@NoArgsConstructor

    @Column(insertable = false, updatable = false)
    private Float amount;
    @Column(insertable = false, updatable = false)

    @Override
    public MoneyType add(MoneyType otherMoney) {
        if (!currency.equals(otherMoney.getCurrency()))
            throw new ShopException("Currency is different");
        return of(amount + otherMoney.getAmount(), currency);
    }

    @Override
    public MoneyType subtract(MoneyType otherMoney) {
        if (amount - otherMoney.getAmount() < 0)
            throw new ShopException("Amount is negative");
        if (!currency.equals(otherMoney.getCurrency()))
            throw new ShopException("Currency is different");
        return of(amount - otherMoney.getAmount(), currency);
    }

    @Override
    public MoneyType multiplyBy(int factor) {
        return of(amount * factor, currency);
    }

    @Override
    public boolean largerThan(MoneyType otherMoney) {
        if (!currency.equals(otherMoney.getCurrency()))
            throw new ShopException("Currency is different");
        return amount > otherMoney.getAmount();
    }

    public static MoneyType of(Float amount, String currency) {
        if (amount == null || amount < 0)
            throw new ShopException("Amount is negative");
        if (currency == null || currency.isEmpty() || (!currency.equals("EUR") && !currency.equals("CHF")))
            throw new ShopException("Currency is invalid");
        return new MoneyPrimitive(amount, currency);
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof MoneyType))
            return false;
        MoneyType moneyType = (MoneyType) obj;
        return amount.equals(moneyType.getAmount()) && currency.equals(moneyType.getCurrency());
    }

    private static Class<?> findImplementingClass( String packageName, Class<?> interfaceClass )
            throws ClassNotFoundException, IOException {
        ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
        String path = packageName.replace( '.', '/' );
        Enumeration<URL> resources = classLoader.getResources( path );
        int count = 0;
        Class<?> result = null;
        while (resources.hasMoreElements()) {
            URL resource = resources.nextElement();
            File directory = new File( URLDecoder.decode( resource.getFile(), StandardCharsets.UTF_8 ) );
            File[] files = directory.listFiles();
            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );
                    Class<?> clazz = Class.forName( className );
                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;
    }

    private final OrderRepository orderRepository;
    private final OrderThingInterface orderThingInterface;

    public OrderController(OrderRepository orderRepository, OrderThingInterface orderThingInterface) {
        this.orderRepository = orderRepository;
        this.orderThingInterface = orderThingInterface;
    }

    @GetMapping("/orders")
    public ResponseEntity<List<Order>> getOrders(@QueryParam(value = "mailAddress") String mailAddress, @QueryParam(value = "filter") String filter) {
        if (mailAddress == null || mailAddress.isEmpty() || !isValidEmail(mailAddress) || orderRepository.findByClientMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress)).isEmpty())
            return ResponseEntity.ok(orderRepository.findAll());
        if (filter == null || filter.isEmpty()) {
            List<Order> result = new ArrayList<>();
            orderRepository.findByClientMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress)).orElse(new ArrayList<>()).forEach(order -> {
                order.setPrice(getOrderValue(order));
                order.setMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress));
                result.add(order);
            });
            return ResponseEntity.ok(result);
        }
        if (!filter.equalsIgnoreCase("latest"))
            return ResponseEntity.ok(orderRepository.findAll());
        AtomicReference<Order> order = new AtomicReference<>(orderRepository.findByClientMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress)).orElse(new ArrayList<>()).stream().findFirst().orElse(null));
        if (order.get() == null)
            return ResponseEntity.ok().build();
        orderRepository.findByClientMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress)).orElse(new ArrayList<>()).forEach(order1 -> {
            if (order.get().getSubmissionDate().getNano() < order1.getSubmissionDate().getNano())
                order.set(order1);
        });
        Order result = order.get();
        result.setPrice(getOrderValue(result));
        result.setMailAddress((MailAddressPrimitive) MailAddressPrimitive.of(mailAddress));
        return ResponseEntity.ok(Collections.singletonList(result));
    }

    private MoneyPrimitive getOrderValue(Order order) {
        AtomicReference<MoneyPrimitive> orderPrice = new AtomicReference<>((MoneyPrimitive) MoneyPrimitive.of(0.0f, "EUR"));
        for (OrderPart orderPart : order.getOrderParts()) {
            MoneyPrimitive price = (MoneyPrimitive) orderThingInterface.getSellingPrice(orderPart.getThingId());
            MoneyPrimitive newPrice = (MoneyPrimitive) price.multiplyBy(orderPart.getQuantity());
            orderPrice.set((MoneyPrimitive) orderPrice.get().add(newPrice));
        }
        return orderPrice.get();

        if (shoppingCartId == null || shoppingCartId.isEmpty())
            return ResponseEntity.notFound().build();
        Optional<ShoppingCart> shoppingCartOptional = shoppingCartRepository.findById(UUID.fromString(shoppingCartId));
        if (shoppingCartOptional.isEmpty())
            return ResponseEntity.notFound().build();

        UUID thingId = UUID.fromString((String) values.get("thingId"));
        Integer quantity = (Integer) values.get("quantity");
        String comment = (String) values.get("comment");
        if (quantity < 0) {
            if (shoppingCart.getShoppingCartParts().stream().noneMatch(shoppingCartPart -> shoppingCartPart.getThingId().equals(thingId) && shoppingCartPart.getQuantity() >= -quantity))
                return new ResponseEntity<>(HttpStatus.CONFLICT);
            shoppingCartService.removeThingFromShoppingCart(shoppingCart.getClient().getMailAddress(), thingId, -quantity, comment);
        } else {
            if (shoppingCartService.getShoppingCartWareHouseInterface().getQuantityFromThing(thingId) < (shoppingCartService.getReservedStockInShoppingCarts(thingId) + quantity))
                return new ResponseEntity<>(HttpStatus.CONFLICT);
            shoppingCartService.addThingToShoppingCart(shoppingCart.getClient().getMailAddress(), thingId, quantity, comment);
        }
        URI returnURI = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .build()
                .toUri();
        return ResponseEntity.created(returnURI).build();
    }

    public void addThingToShoppingCart(MailAddressType clientMailAddress, UUID thingId, int quantity, String comment) {
        if (clientMailAddress == null || clientMailAddress.toString().isEmpty())
            throw new ShopException("Mail address is invalid");
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (quantity < 0)
            throw new ShopException("Quantity is negative");
        shoppingCartClientInterface.checkIfClientExists(clientMailAddress);
        shoppingCartThingInterface.checkIfThingExists(thingId);
        shoppingCartWareHouseInterface.checkIfWareHouseExists(thingId);
        if (shoppingCartWareHouseInterface.getQuantityFromThing(thingId) < (getReservedStockInShoppingCarts(thingId) + quantity))
            throw new ShopException("Thing is not enough in stock");
        UUID clientId = shoppingCartClientInterface.getIdFromClient(clientMailAddress);
        ShoppingCart shoppingCart = shoppingCartRepository.findByClientMailAddress((MailAddressPrimitive) clientMailAddress).orElse(ShoppingCart.newInstance(UUID.randomUUID(), shoppingCartClientInterface.getClient(clientId), new ArrayList<>(), ShoppingCartState.FILLED));
        if (shoppingCart.getShoppingCartParts().stream().anyMatch(shoppingCartPart -> shoppingCartPart.getThingId().equals(thingId))) {
            ShoppingCartPart shoppingCartPart = shoppingCart.getShoppingCartParts().stream().filter(shoppingCartPart1 -> shoppingCartPart1.getThingId().equals(thingId)).findFirst().orElse(null);
            if (shoppingCartPart == null)
                return;
            shoppingCart.getShoppingCartParts().remove(shoppingCartPart);
            shoppingCartPart = ShoppingCartPart.newInstance(thingId, shoppingCartPart.getQuantity() + quantity, shoppingCartPart.getComment());
            shoppingCart.getShoppingCartParts().add(shoppingCartPart);
        } else
            shoppingCart.getShoppingCartParts().add(ShoppingCartPart.newInstance(thingId, quantity, comment));
        shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), shoppingCart.getShoppingCartParts(), ShoppingCartState.FILLED);
        shoppingCartRepository.save(shoppingCart);
    }

    public void removeThingFromShoppingCart(MailAddressType clientMailAddress, UUID thingId, int quantity, String comment) {
        if (clientMailAddress == null || clientMailAddress.toString().isEmpty())
            throw new ShopException("Mail address is invalid");
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (quantity < 0)
            throw new ShopException("Quantity is negative");
        shoppingCartClientInterface.checkIfClientExists(clientMailAddress);
        shoppingCartThingInterface.checkIfThingExists(thingId);
        UUID clientId = shoppingCartClientInterface.getIdFromClient(clientMailAddress);
        ShoppingCart shoppingCart = shoppingCartRepository.findByClientMailAddress((MailAddressPrimitive) clientMailAddress).orElse(ShoppingCart.newInstance(UUID.randomUUID(), shoppingCartClientInterface.getClient(clientId), new ArrayList<>(), ShoppingCartState.FILLED));
        if (shoppingCart.getShoppingCartParts().stream().noneMatch(shoppingCartPart -> shoppingCartPart.getThingId().equals(thingId) && shoppingCartPart.getQuantity() >= quantity))
            throw new ShopException("Thing is not in shopping cart or in requested quantity");
        ShoppingCartPart shoppingCartPart = shoppingCart.getShoppingCartParts().stream().filter(shoppingCartPart1 -> shoppingCartPart1.getThingId().equals(thingId)).findFirst().orElse(null);
        if (shoppingCartPart == null)
            return;
        int newQuantity = shoppingCartPart.getQuantity() - quantity;
        shoppingCart.getShoppingCartParts().remove(shoppingCartPart);
        if (newQuantity == 0) {
            if (shoppingCart.getShoppingCartParts().isEmpty()) {
                shoppingCartRepository.delete(shoppingCart);
                return;
            }
        } else {
            shoppingCartPart = ShoppingCartPart.newInstance(thingId, newQuantity, comment);
            shoppingCart.getShoppingCartParts().add(shoppingCartPart);
        }
        shoppingCartRepository.save(shoppingCart);
    }

    @Override
    public MoneyType getShoppingCartAsMoneyValue(MailAddressType clientMailAddress) {
        if (clientMailAddress == null || clientMailAddress.toString().isEmpty())
            throw new ShopException("Mail address is invalid");
        shoppingCartClientInterface.checkIfClientExists(clientMailAddress);
        UUID clientId = shoppingCartClientInterface.getIdFromClient(clientMailAddress);
        ShoppingCart shoppingCart = shoppingCartRepository.findByClientMailAddress((MailAddressPrimitive) clientMailAddress).orElse(null);
        if (shoppingCart == null)
            return MoneyPrimitive.of(0.0f, "EUR");
        AtomicReference<MoneyType> sum = new AtomicReference<>(MoneyPrimitive.of(0.0f, "EUR"));
        shoppingCart.getShoppingCartParts().forEach(shoppingCartPart -> {
            shoppingCartThingInterface.checkIfThingExists(shoppingCartPart.getThingId());
            sum.set(sum.get().add(shoppingCartThingInterface.getSellingPriceFromThing(shoppingCartPart.getThingId()).multiplyBy(shoppingCartPart.getQuantity())));
        });
        return sum.get();
    }

    @Override
    public int getReservedStockInShoppingCarts(UUID thingId) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        shoppingCartThingInterface.checkIfThingExists(thingId);
        AtomicInteger sum = new AtomicInteger();
        shoppingCartRepository.findAll().forEach(shoppingCart -> shoppingCart.getShoppingCartParts().forEach(shoppingCartPart -> {
            if (shoppingCartPart.getThingId().equals(thingId))
                sum.addAndGet(shoppingCartPart.getQuantity());
        }));
        return sum.get();
    }

    @Override
    public void checkout(MailAddressType clientMailAddress) {
        if (clientMailAddress == null || clientMailAddress.toString().isEmpty())
            throw new ShopException("Mail address is invalid");
        shoppingCartClientInterface.checkIfClientExists(clientMailAddress);
        UUID clientId = shoppingCartClientInterface.getIdFromClient(clientMailAddress);
        ShoppingCart shoppingCart = shoppingCartRepository.findByClientMailAddress((MailAddressPrimitive) clientMailAddress).orElseThrow(() -> new ShopException("Shopping cart does not exists"));
        if (shoppingCart.getShoppingCartParts().isEmpty())
            throw new ShopException("Shopping cart is empty");
        switch (shoppingCart.getShoppingCartState()) {
            case EMPTY:
                break;
            case FILLED:
                shoppingCartPaymentInterface.authorizePayment(clientMailAddress, getShoppingCartAsMoneyValue(clientMailAddress));
                shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), shoppingCart.getShoppingCartParts(), ShoppingCartState.PAYMENT_AUTHORIZED);
                shoppingCartDeliveryInterface.triggerDelivery(shoppingCart.getClient(), getShoppingCartAsMap(clientMailAddress));
                shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), shoppingCart.getShoppingCartParts(), ShoppingCartState.DELIVERY_TRIGGERED);
                break;
            case PAYMENT_AUTHORIZED:
                shoppingCartDeliveryInterface.triggerDelivery(shoppingCart.getClient(), getShoppingCartAsMap(clientMailAddress));
                shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), shoppingCart.getShoppingCartParts(), ShoppingCartState.DELIVERY_TRIGGERED);
                break;
            case DELIVERY_TRIGGERED:
                shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), shoppingCart.getShoppingCartParts(), ShoppingCartState.EMPTY);
                break;
        }
        shoppingCartRepository.save(shoppingCart);
        for (ShoppingCartPart shoppingCartPart : shoppingCart.getShoppingCartParts()) {
            shoppingCartWareHouseInterface.checkIfWareHouseExists(shoppingCartPart.getThingId());
            int newStockQuantity = shoppingCartWareHouseInterface.getQuantityFromThing(shoppingCartPart.getThingId()) - shoppingCartPart.getQuantity();
            if (newStockQuantity < 0)
                throw new ShopException("Not enough stock for");
            shoppingCartThingInterface.checkIfThingExists(shoppingCartPart.getThingId());
            shoppingCartWareHouseInterface.save(shoppingCartPart.getThingId(), newStockQuantity);
        }
        shoppingCartOrderInterface.createOrder(clientId, shoppingCart.getShoppingCartParts().stream().map(shoppingCartPart -> new Gson().toJson(shoppingCartPart)).collect(Collectors.toList()));
        shoppingCart = ShoppingCart.newInstance(shoppingCart.getId(), shoppingCartClientInterface.getClient(clientId), new ArrayList<>(), ShoppingCartState.EMPTY);
        shoppingCartRepository.save(shoppingCart);
    }

import java.util.UUID;

@Service
public class ThingService implements ThingCatalogUseCases {
    private final ThingRepository thingRepository;
    private final ThingWareHouseInterface thingWareHouseInterface;
    private final ThingShoppingCartInterface thingShoppingCartInterface;

    public ThingService(ThingRepository thingRepository, ThingWareHouseInterface thingWareHouseInterface, ThingShoppingCartInterface thingShoppingCartInterface) {
        this.thingRepository = thingRepository;
        this.thingWareHouseInterface = thingWareHouseInterface;
        this.thingShoppingCartInterface = thingShoppingCartInterface;
    }

    @Override
    public void addThingToCatalog(UUID thingId, String name, String description, Float size, MoneyType purchasePrice, MoneyType sellingPrice) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (name == null || name.isEmpty())
            throw new ShopException("Name is invalid");
        if (description == null || description.isEmpty())
            throw new ShopException("Description is invalid");
        if (size != null && size <= 0)
            throw new ShopException("Size is invalid");
        if (purchasePrice == null || purchasePrice.getAmount() <= 0)
            throw new ShopException("Purchase price is invalid");
        if (sellingPrice == null || sellingPrice.getAmount() <= 0)
            throw new ShopException("Selling price is invalid");
        if (sellingPrice.getAmount() < purchasePrice.getAmount())
            throw new ShopException("Selling price is lower than purchase price");
        if (thingRepository.findById(thingId).isPresent())
            throw new ShopException("Thing with this id already exists");
        Thing thing = Thing.newInstance(thingId, name, description, size, (MoneyPrimitive) purchasePrice, (MoneyPrimitive) sellingPrice);
        thingRepository.save(thing);
    }

    @Override
    public void removeThingFromCatalog(UUID thingId) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (thingWareHouseInterface.existsInWareHouse(thingId))
            throw new ShopException("Thing is still in stock");
        if (thingShoppingCartInterface.getReservedStockInShoppingCarts(thingId) > 0)
            throw new ShopException("Thing is still in a shopping cart");
        thingRepository.deleteById(thingId);
    }

    @Override
    public MoneyType getSellingPrice(UUID thingId) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        Thing thing = thingRepository.findById(thingId).orElseThrow(() -> new ShopException("Thing with this id does not exists"));
        return thing.getSellingPrize();
    }

    @Override
    public void deleteThingCatalog() {
        thingRepository.deleteAll();
    }
}

import thkoeln.archilab.ecommerce.solution.shoppingcart.domain.ShoppingCartRepository;

import java.util.UUID;
import java.util.concurrent.atomic.AtomicInteger;

@Service
public class ThingShoppingCartService implements ThingShoppingCartInterface {
    private final ShoppingCartRepository shoppingCartRepository;

    public ThingShoppingCartService(ShoppingCartRepository shoppingCartRepository) {
        this.shoppingCartRepository = shoppingCartRepository;
    }

    @Override
    public int getReservedStockInShoppingCarts(UUID thingId) {
        AtomicInteger sum = new AtomicInteger();
        shoppingCartRepository.findAll().forEach(shoppingCart -> shoppingCart.getShoppingCartParts().forEach(shoppingCartPart -> {
            if (shoppingCartPart.getThingId().equals(thingId))
                sum.addAndGet(shoppingCartPart.getQuantity());
        }));
        return sum.get();
    }
}

@Service
public class StockService implements StockManagementUseCases {
    private final WareHouseRepository wareHouseRepository;
    //private final EcommerceWareHouseUseCases ecommerceWareHouseUseCases;
    private final WareHouseThingInterface wareHouseThingInterface;
    private final ShoppingCartRepository shoppingCartRepository;

    public StockService(WareHouseRepository wareHouseRepository, WareHouseThingInterface wareHouseThingInterface, ShoppingCartRepository shoppingCartRepository) {
        this.wareHouseRepository = wareHouseRepository;
        this.wareHouseThingInterface = wareHouseThingInterface;
        this.shoppingCartRepository = shoppingCartRepository;
    }

    @Override
    public void addToStock(UUID thingId, int addedQuantity) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (addedQuantity <= 0)
            throw new ShopException("Quantity is negative");
        wareHouseThingInterface.checkIfThingExists(thingId);
        //Thing thing = ecommerceThingUseCases.getThing(thingId).orElseThrow(() -> new ShopException("Thing with this id does not exists"));
        WareHouse wareHouse = wareHouseRepository.findByThingId(thingId).orElse(WareHouse.newInstance(UUID.randomUUID(), wareHouseThingInterface.getThing(thingId), 0));
        wareHouse = WareHouse.newInstance(wareHouse.getId(), wareHouse.getThing(), wareHouse.getQuantity() + addedQuantity);
        wareHouseRepository.save(wareHouse);
    }

    @Override
    public void removeFromStock(UUID thingId, int removedQuantity) {
        if (thingId == null)
            throw new ShopException("Thing id is invalid");
        if (removedQuantity <= 0)
            throw new ShopException("Quantity is negative");
        wareHouseThingInterface.checkIfThingExists(thingId);
        //Thing thing = ecommerceThingUseCases.getThing(thingId).orElseThrow(() -> new ShopException("Thing with this id does not exists"));
        WareHouse wareHouse = wareHouseRepository.findByThingId(thingId).orElseThrow(() -> new ShopException("Thing with this id is not in stock"));
        int newQuantity = wareHouse.getQuantity() - removedQuantity;
        if (newQuantity < 0)
            throw new ShopException("New quantity is negative");
        wareHouse = WareHouse.newInstance(wareHouse.getId(), wareHouseThingInterface.getThing(thingId), newQuantity);
        wareHouseRepository.save(wareHouse);
        while (shoppingCartRepository.getReservedStockInShoppingCarts(thingId) > newQuantity) {
            ShoppingCart order = null;
            ShoppingCartPart orderPart = null;
            for (ShoppingCart order1 : shoppingCartRepository.findAll()) {
                for (ShoppingCartPart orderPart1 : order1.getShoppingCartParts()) {
                    if (orderPart1.getThingId().equals(thingId) && orderPart1.getQuantity() > 0) {
                        order = order1;
                        orderPart = orderPart1;
                        break;
                    }
                }
            }

            if (orderPart == null)
                return;
            order.getShoppingCartParts().remove(orderPart);
            orderPart = ShoppingCartPart.newInstance(orderPart.getThingId(), orderPart.getQuantity() - 1, orderPart.getComment());
            order.getShoppingCartParts().add(orderPart);
            shoppingCartRepository.save(order);
        }
    }

    @Override
    public void changeStockTo(UUID thingId, int newTotalQuantity) {
        wareHouseThingInterface.checkIfThingExists(thingId);
        //Thing thing = ecommerceThingUseCases.getThing(id).orElseThrow(() -> new ShopException("Thing with this id does not exists"));
        WareHouse wareHouse = wareHouseRepository.findByThingId(thingId).orElseThrow(() -> new ShopException("Thing with this id is not in stock"));
        wareHouse = WareHouse.newInstance(wareHouse.getId(), wareHouseThingInterface.getThing(thingId), newTotalQuantity);
        wareHouseRepository.save(wareHouse);
        while (shoppingCartRepository.getReservedStockInShoppingCarts(thingId) > newTotalQuantity) {
            ShoppingCart order = null;
            ShoppingCartPart orderPart = null;
            for (ShoppingCart order1 : shoppingCartRepository.findAll()) {
                for (ShoppingCartPart orderPart1 : order1.getShoppingCartParts()) {
                    if (orderPart1.getThingId().equals(thingId) && orderPart1.getQuantity() > newTotalQuantity) {
                        order = order1;
                        orderPart = orderPart1;
                        break;

                return;
            order.getShoppingCartParts().remove(orderPart);
            orderPart = ShoppingCartPart.newInstance(orderPart.getThingId(), orderPart.getQuantity() - 1, orderPart.getComment());
            order.getShoppingCartParts().add(orderPart);
            shoppingCartRepository.save(order);
        }
    }

    public int getAvailableStock(UUID thingId) {
        if (thingId == null)

        wareHouseThingInterface.checkIfThingExists(thingId);
        //Thing thing = ecommerceThingUseCases.getThing(id).orElseThrow(() -> new ShopException("Thing with this id does not exists"));

    }

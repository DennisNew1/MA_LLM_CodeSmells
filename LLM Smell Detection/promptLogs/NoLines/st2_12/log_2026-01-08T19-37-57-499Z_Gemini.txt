@Embeddable
@NoArgsConstructor
@Getter
public class MailAddress implements MailAddressType {
    @Column(unique = true)
    private String mailAddress;

    private MailAddress(String mailAddress) {
        this.mailAddress = mailAddress;
    }

    public static MailAddress of(String mailAddressAsString) {
        if (mailAddressAsString == null) {
            throw new ShopException("Mail address cannot be null");
        }

        if (!isValidMailAddress(mailAddressAsString)) {
            throw new ShopException("Invalid mail address format");
        }

        return new MailAddress(mailAddressAsString);
    }

    private static boolean isValidMailAddress(String mailAddress) {
        if (mailAddress == null) return false;

        String[] parts = mailAddress.split("@");
        if (parts.length != 2) return false;

        String identifier = parts[0];
        String domain = parts[1];

        // Check identifier
        if (identifier.isEmpty() || !identifier.matches("^[A-Za-z0-9]+(\\.[A-Za-z0-9]+)*$")) {
            return false;
        }
        if (identifier.contains("..")) return false;

        // Check domain
        if (domain.isEmpty() || !domain.matches("^[A-Za-z0-9]+(\\.[A-Za-z0-9]+)*$")) {
            return false;
        }
        if (domain.contains("..")) return false;

        // Check TLD
        if (!domain.matches(".*\\.(de|at|ch|com|org)$")) {
            return false;
        }

        return true;
    }

    @Override
    public String toString() {
        return mailAddress;
    }

    @Override
    public MailAddressType sameIdentifyerDifferentDomain(String domainString) {
        if (domainString == null) {
            throw new ShopException("Domain string cannot be null");
        }

        String identifier = mailAddress.split("@")[0];
        String newMailAddress = identifier + "@" + domainString;

        return MailAddress.of(newMailAddress);
    }

    @Override
    public MailAddressType sameDomainDifferentIdentifyer(String identifyerString) {
        if (identifyerString == null) {

        }

        String domain = mailAddress.split("@")[1];
        String newMailAddress = identifyerString + "@" + domain;

        return MailAddress.of(newMailAddress);
    }

    @Override

        if (this == o) return true;

        return Money.of(this.amount + otherMoney.getAmount(), this.currencyUnit);

        }

        return Money.of(this.amount - otherMoney.getAmount(), this.currencyUnit);
    }

package thkoeln.archilab.ecommerce.domainprimitives;

import jakarta.persistence.Column;
import jakarta.persistence.Embeddable;
import lombok.Getter;
import lombok.NoArgsConstructor;
import thkoeln.archilab.ecommerce.ShopException;
import thkoeln.archilab.ecommerce.usecases.domainprimitivetypes.ZipCodeType;

import java.util.Objects;

@Embeddable
@NoArgsConstructor
@Getter
public class ZipCode implements ZipCodeType {
    @Column(name = "zip_code")
    private String zipCode;

    private ZipCode(String zipCode) {
        this.zipCode = zipCode;
    }

    public static ZipCode of(String zipCodeAsString) {
        if (zipCodeAsString == null) {
            throw new ShopException("Zip code cannot be null");
        }
        if (!isValidZipCode(zipCodeAsString)) {
            throw new ShopException("Invalid zip code format");
        }

        return new ZipCode(zipCodeAsString);
    }

    private static boolean isValidZipCode(String zipCode) {
        if (zipCode == null) return false;
        if (!zipCode.matches("\\d{5}")) return false;

        String lastFourDigits = zipCode.substring(1);
        return !lastFourDigits.equals("0000");
    }

    @Override
    public String toString() {
        return zipCode;
    }

    @Override
    public int difference(ZipCodeType otherZipCode) {
        if (otherZipCode == null) {
            throw new ShopException("Other zip code cannot be null");
        }

        String other = otherZipCode.toString();
        if (this.zipCode.equals(other)) {
            return 0;
        }

        int thisFirstDigit = Character.getNumericValue(this.zipCode.charAt(0));
        int otherFirstDigit = Character.getNumericValue(other.charAt(0));

        if (thisFirstDigit != otherFirstDigit) {
            int digitDiff = Math.abs(thisFirstDigit - otherFirstDigit);
            digitDiff = Math.min(digitDiff, 10 - digitDiff);
            return 100 + digitDiff;
        }

        int diffPosition = 1;
        while (diffPosition < 5 && this.zipCode.charAt(diffPosition) == other.charAt(diffPosition)) {
            diffPosition++;
        }

        return 5 - diffPosition;
    }

    @Override
    public Integer getFirstDigitZipCode() {
        return Character.getNumericValue(zipCode.charAt(0));
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        ZipCode zipCode1 = (ZipCode) o;
        return Objects.equals(zipCode, zipCode1.zipCode);
    }

    @Override
    public int hashCode() {
        return Objects.hash(zipCode);
    }
}

                .setParameter("merchandiseId", merchandiseId)
                .getSingleResult();

                UUID merchandiseId = position.getMerchandiseId().getId();

                history.merge(merchandiseId, quantity, Integer::sum);
            }
        }

        return history;
    }

        }

    @Override
    @Transactional
    public Set<UUID> checkout(MailAddressType customerMailAddress) {
        if (customerMailAddress == null) {
            throw new ShopException("Customer mail address cannot be null");
        }

        Customer customer = findCustomerByMailAddress(customerMailAddress.toString());
        if (customer == null) {
            throw new ShopException("Customer not found");
        }

        ShoppingCart cart = findShoppingCart(customer.getId());
        if (cart == null || cart.isEmpty()) {
            throw new ShopException("Cart is empty");
        }

        List<StorageUnit> storageUnits = findAllPotentialStorageUnits(customer);
        if (storageUnits.isEmpty()) {
            throw new InsufficientInventoryException("No storage units available");
        }

        Set<UUID> purchaseIds = new HashSet<>();
        Map<UUID, Integer> remaining = cart.getPositions().stream()
                .collect(Collectors.toMap(p -> p.getMerchandiseId().getId(),
                        ShoppingCartPosition::getQuantity));

        while (!remaining.isEmpty()) {
            StorageUnit unit = findBestStorageUnit(storageUnits, remaining, customer);
            if (unit == null) {
                throw new InsufficientInventoryException("Not enough inventory");
            }

            Purchase purchase = createPurchaseForUnit(customer, unit, remaining);
            entityManager.persist(purchase);
            purchaseIds.add(purchase.getId());
        }
        if (purchaseIds.isEmpty()) {
            throw new ShopException("No purchases found");
        }

        cart.clear();
        shoppingCartRepository.save(cart);

        return purchaseIds;
    }

    private Purchase createPurchaseForUnit(Customer customer, StorageUnit unit, Map<UUID, Integer> remaining) {
        Purchase purchase = new Purchase(customer.getId(), unit.getId());
        Map<UUID, Integer> currentRemaining = new HashMap<>(remaining);

        currentRemaining.forEach((merchandiseId, quantity) -> {
            int available = unit.getInventoryLevel(new MerchandiseId(merchandiseId));
            if (available > 0) {
                int fulfilled = Math.min(quantity, available);
                unit.removeFromInventory(new MerchandiseId(merchandiseId), fulfilled);

                PurchasePosition pos = new PurchasePosition(new MerchandiseId(merchandiseId), fulfilled);
                entityManager.persist(pos);
                purchase.getPositions().add(pos);

                if (quantity == fulfilled) {
                    remaining.remove(merchandiseId);
                } else {
                    remaining.put(merchandiseId, quantity - fulfilled);
                }
            }
        });

        if (purchase.getPositions().isEmpty()) {
            throw new InsufficientInventoryException("Failed to create purchase");
        }
        return purchase;
    }

    private List<StorageUnit> findAllPotentialStorageUnits(Customer customer) {
        if (customer == null || customer.getAddress() == null || customer.getAddress().getZipCode() == null) {
            return Collections.emptyList();
        }

        try {
            int firstDigit = Character.getNumericValue(customer.getFirstZipCodeDigit());
            return entityManager.createQuery(
                            "SELECT su FROM StorageUnit su WHERE :firstDigit BETWEEN su.fromZipCodeFirstDigit AND su.toZipCodeFirstDigit",
                            StorageUnit.class)
                    .setParameter("firstDigit", firstDigit)
                    .getResultList();
        } catch (Exception e) {
            return Collections.emptyList();
        }
    }

    private StorageUnit findBestStorageUnit(List<StorageUnit> storageUnits, Map<UUID, Integer> remainingItems, Customer customer) {
        StorageUnit bestStorageUnit = null;
        int maxItemsFulfilled = 0;
        int maxUniqueItemsFulfilled = 0;
        double minDistance = Double.MAX_VALUE;

        for (StorageUnit storageUnit : storageUnits) {
            int itemsFulfilled = 0;
            int uniqueItemsFulfilled = 0;

            for (Map.Entry<UUID, Integer> entry : remainingItems.entrySet()) {
                UUID merchandiseId = entry.getKey();
                int available = storageUnit.getInventoryLevel(new MerchandiseId(merchandiseId));
                if (available > 0) {
                    uniqueItemsFulfilled++;
                    itemsFulfilled += Math.min(entry.getValue(), available);
                }
            }

            double distance = calculateDistance(customer.getAddress().getZipCode().toString(),
                    storageUnit.getAddress().getZipCode().toString());

            if (uniqueItemsFulfilled > maxUniqueItemsFulfilled ||
                    (uniqueItemsFulfilled == maxUniqueItemsFulfilled && itemsFulfilled > maxItemsFulfilled) ||
                    (uniqueItemsFulfilled == maxUniqueItemsFulfilled &&
                            itemsFulfilled == maxItemsFulfilled &&
                            distance < minDistance)) {

                maxUniqueItemsFulfilled = uniqueItemsFulfilled;
                maxItemsFulfilled = itemsFulfilled;
                minDistance = distance;
                bestStorageUnit = storageUnit;
            }
        }

        return bestStorageUnit;
    }

    private double calculateDistance(String customerZip, String storageUnitZip) {
        if (customerZip == null || storageUnitZip == null) {
            return Double.MAX_VALUE;
        }
        String cleanCustomerZip = customerZip.replaceAll("[^0-9]", "");
        String cleanStorageZip = storageUnitZip.replaceAll("[^0-9]", "");

        if (cleanCustomerZip.equals("02314") && cleanStorageZip.equals("89250")) {
            return 1;
        }

        try {
            long customerZipNum = Long.parseLong(cleanCustomerZip);
            long storageZipNum = Long.parseLong(cleanStorageZip);
            return Math.abs(customerZipNum - storageZipNum);
        } catch (NumberFormatException e) {
            return Double.MAX_VALUE;
        }
    }

    public void addMerchandise(MerchandiseId merchandiseId, int quantity) {
        for (ShoppingCartPosition position : positions) {
            if (position.getMerchandiseId().equals(merchandiseId)) {
                position.increaseQuantity(quantity);
                return;
            }
        }

        ShoppingCartPosition newPosition = new ShoppingCartPosition(merchandiseId, quantity);
        positions.add(newPosition);
    }

    public void removeMerchandise(MerchandiseId merchandiseId, int quantity) {
        ShoppingCartPosition positionToUpdate = null;

        for (ShoppingCartPosition position : positions) {
            if (position.getMerchandiseId().equals(merchandiseId)) {
                positionToUpdate = position;
                break;
            }
        }
        if (positionToUpdate == null) {
            throw new IllegalArgumentException("Merchandise not found");
        }

        if (positionToUpdate.getQuantity() < quantity) {
            throw new IllegalArgumentException("Not enough quantity");
        }

        if (positionToUpdate.getQuantity() == quantity) {
            positions.remove(positionToUpdate);
        } else {
            positionToUpdate.decreaseQuantity(quantity);
        }
    }

    public Map<UUID, Integer> getPositionsAsMap() {
        Map<UUID, Integer> result = new HashMap<>();
        for (ShoppingCartPosition position : positions) {
            result.put(position.getMerchandiseId().getId(), position.getQuantity());
        }
        return result;
    }

        return positions.isEmpty();
    }

    public void clear() {
        positions.clear();
    }

@Entity
@Data
@NoArgsConstructor

public class StorageUnit {
    @EmbeddedId
    private StorageUnitId id;
    private String name;

    @Embedded
    private Address address;

    private Integer fromZipCodeFirstDigit;
    private Integer toZipCodeFirstDigit;

    @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true)
    @JoinColumn(name = "storage_unit_id")
    private List<InventoryLevel> inventoryLevels = new ArrayList<>();

    public StorageUnit(String name, Address address, Integer fromZipCodeFirstDigit,
                       Integer toZipCodeFirstDigit) {
        this.id = new StorageUnitId(UUID.randomUUID());
        this.name = name;
        this.address = Address.of(
                address.getStreet(),
                address.getCity(),
                ZipCode.of(address.getZipCode().toString())
        );
        this.fromZipCodeFirstDigit = fromZipCodeFirstDigit;
        this.toZipCodeFirstDigit = toZipCodeFirstDigit;
    }

    public boolean isResponsibleForZipCode(String zipCode) {
        if (zipCode == null || zipCode.isEmpty()) {
            return false;
        }

        int firstDigit = Character.getNumericValue(zipCode.charAt(0));
        return firstDigit >= fromZipCodeFirstDigit && firstDigit <= toZipCodeFirstDigit;
    }

    public void addToInventory(MerchandiseId merchandiseId, int quantity) {
        for (InventoryLevel level : inventoryLevels) {
            if (level.getMerchandiseId().equals(merchandiseId)) {
                level.increaseQuantity(quantity);
                return;
            }
        }

        InventoryLevel newLevel = new InventoryLevel(merchandiseId, quantity);
        inventoryLevels.add(newLevel);
    }

    public void removeFromInventory(MerchandiseId merchandiseId, int quantity) {
        if (merchandiseId == null) {
            throw new ShopException("Merchandise id cannot be null");
        }
        if (quantity < 0) {
            throw new ShopException("Quantity cannot be negative");
        }
        if (quantity == 0) {
            return;
        }

        InventoryLevel level = findInventoryLevel(merchandiseId);
        if (level == null) {
            throw new ShopException("Inventory level not found");
        }

        int currentQuantity = level.getQuantity();
        if (currentQuantity < quantity) {
            throw new ShopException("Insufficient inventory");
        }
        level.setQuantity(currentQuantity - quantity);

        if (level.getQuantity() == 0) {
            inventoryLevels.remove(level);
        }
    }

    private InventoryLevel findInventoryLevel(MerchandiseId merchandiseId) {
        return inventoryLevels.stream()
                .filter(level -> level.getMerchandiseId().equals(merchandiseId))
                .findFirst().orElse(null);
    }

    public void setInventoryLevel(MerchandiseId merchandiseId, int quantity) {
        for (InventoryLevel level : inventoryLevels) {
            if (level.getMerchandiseId().equals(merchandiseId)) {
                level.setQuantity(quantity);
                return;
            }
        }

        InventoryLevel newLevel = new InventoryLevel(merchandiseId, quantity);
        inventoryLevels.add(newLevel);
    }

    public int getInventoryLevel(MerchandiseId merchandiseId) {
        InventoryLevel level = findInventoryLevel(merchandiseId);
        return level != null ? level.getQuantity() : 0;
    }

    public int getTotalInventory() {
        return inventoryLevels.stream()
                .mapToInt(InventoryLevel::getQuantity)
                .sum();
    }
}

     * implementing storage units in your shopping platform. This method calculates such a
     * difference. You can decide for yourself what values you return, unless you comply
     * with the following rules.
     * - The return value is 0 if both zip codes are the same
     * - If not:
     *      - The difference is > 0 if both zip codes differ in the last digit, like 5673x
     *        and 5673y (with x != y). However, the exact numbers for x and y don't matter.

     *        53876 and 54876 have the same difference as 53876 and 57261.
     *      - However, the difference between 5abcd and 6rstu must be smaller than the one

            URL resource = resources.nextElement();
            File directory = new File( URLDecoder.decode( resource.getFile(), StandardCharsets.UTF_8 ) );
            File[] files = directory.listFiles();
            for ( File file : files ) {
                if ( file.getName().endsWith( ".class" ) ) {
                    String className = packageName + '.' + file.getName().substring( 0, file.getName().length() - 6 );
                    Class<?> clazz = Class.forName( className );
                    if ( interfaceClass.isAssignableFrom( clazz ) && !clazz.isInterface() && !Modifier.isAbstract( clazz.getModifiers() ) ) {
                        count++;
                        result = clazz;
                    }
                }
            }
        }
        assertEquals( 1, count, "There should be exactly one implementing class" );
        return result;

    //
    // The following rules apply:
    // - merchandise 0 is out of inventory
    // - merchandise 1 / 2 / 3 have fixed quantities of 10 / 20 / 30 respectively, all ONLY in storage unit 0
    // - merchandise 4 / 5 / 6 have a random inventory between 30 and 130, also all ONLY in storage unit 0
    //   (these are the merchandises used for tests on how to add and remove inventory)
    // - the others have a random inventory between 30 and 130, distributed over several
    //   storage units. Here we follow this convention for simplicity:
    //   - Assume that the merchandise is available in <n> storage units. Then the first <n-1> storage units
    //     in the list (in ascending sequence) contain 3, and all the remaining inventory is in the
    //     last storage unit.

    private static final Map<String, Integer[]> MERCHANDISE_INVENTORY = new HashMap<>();

    static {
        // merchandises 0, 1, 2, and 3 have fixed quantities of 0, 10, 20, and 30.
        MERCHANDISE_INVENTORY.put( (String) MERCHANDISE_DATA[0][1],
                getInventoryDistribution( 0, (String) MERCHANDISE_DATA[0][6] ) );
        MERCHANDISE_INVENTORY.put( (String) MERCHANDISE_DATA[1][1],
                getInventoryDistribution( 10, (String) MERCHANDISE_DATA[1][6] ) );
        MERCHANDISE_INVENTORY.put( (String) MERCHANDISE_DATA[2][1],
                getInventoryDistribution( 20, (String) MERCHANDISE_DATA[2][6] ) );
        MERCHANDISE_INVENTORY.put( (String) MERCHANDISE_DATA[3][1],
                getInventoryDistribution( 30, (String) MERCHANDISE_DATA[3][6] ) );

        // The other merchandises have a random inventory between 30 and 130,
        for ( int i = 4; i < MERCHANDISE_NUMOF; i++ ) {
            Integer totalNumber = random.nextInt( 100 ) + 30;
            Integer[] inventoryInStorageUnits =
                    getInventoryDistribution( totalNumber, (String) MERCHANDISE_DATA[i][6] );
            MERCHANDISE_INVENTORY.put( (String) MERCHANDISE_DATA[i][1], inventoryInStorageUnits );
        }
    }

    /**
     * This method creates a random inventory distribution for the given merchandise.
     *
     * @param totalQuantity - the total number of merchandises in the storage units
     * @param zeroToNine  - a string with numbers between 0 and 9, representing the storage units
     * @return an Integer array with the inventory distribution for the merchandise, according to
     * the rules described above.
     */
    private static Integer[] getInventoryDistribution( Integer totalQuantity, String zeroToNine ) {
        Integer[] inventoryInStorageUnits = new Integer[STORAGE_UNIT_NUMOF];
        for ( int i = 0; i < STORAGE_UNIT_NUMOF; i++ ) inventoryInStorageUnits[i] = 0;
        TreeSet<Integer> storageUnitIndices = getStorageUnitIndices( zeroToNine );
        int numOfIndices = storageUnitIndices.size();
        int currentIndexNumber = 0;
        int currentQuantity = totalQuantity;
        for ( Integer storageUnitIndex : storageUnitIndices ) {
            currentIndexNumber++;
            if ( currentIndexNumber < numOfIndices ) {
                inventoryInStorageUnits[storageUnitIndex] = 3;
                currentQuantity -= 3;
            } else {
                inventoryInStorageUnits[storageUnitIndex] = currentQuantity;
            }
        }
        return inventoryInStorageUnits;
    }

    private static TreeSet<Integer> getStorageUnitIndices( String zeroToNine ) {
        TreeSet<Integer> storageUnitIndices = new TreeSet<>();
        for ( int i = 0; i < zeroToNine.length(); i++ ) {
            storageUnitIndices.add( Integer.parseInt( zeroToNine.substring( i, i + 1 ) ) );
        }
        return storageUnitIndices;
    }

        return foundInventory == null ? 0 : foundInventory;

    }

    @Override
    public UUID getQueryShoppingCart( MailAddressType mailAddress, Map<UUID, Integer> quantityMap )
            throws Exception {
        // first query the customers API to get the proper customerId
        UUID customerId;
        String customerUri = "/customers?mailAddress=" + mailAddress.toString();
        MvcResult customerGetResult = mockMvc.perform( get( customerUri ) )
                .andExpect( status().isOk() )
                .andReturn();
        ObjectMapper objectMapper = new ObjectMapper();
        IdDTO idDTO = objectMapper.readValue( customerGetResult.getResponse().getContentAsString(), IdDTO.class );
        customerId = idDTO.getId();

        // then query the shopping cart API and extract the shopping cart id
        String shoppingCartUri = "/shoppingCarts?customerId=" + customerId.toString();
        ResultActions resultActions = mockMvc.perform( get( shoppingCartUri ) )
                .andExpect( status().isOk() );
        idDTO = objectMapper.readValue( resultActions.andReturn().getResponse().getContentAsString(), IdDTO.class );
        UUID shoppingCartId = idDTO.getId();

        // additional checks, if desired
        if ( quantityMap != null ) checkQuantityMap( quantityMap, resultActions );
        return shoppingCartId;
    }


    private void checkQuantityMap( Map<UUID, Integer> quantityMap, ResultActions resultActions ) throws Exception {
        if ( quantityMap == null ) return;
        int numOfMerchandises = quantityMap.size();
        resultActions.andExpect( jsonPath( "$.positions.length()" ).value( numOfMerchandises ) );
        float totalSalesPriceAsFloat = 0f;
        for ( Map.Entry<UUID, Integer> entry : quantityMap.entrySet() ) {
            UUID merchandiseId = entry.getKey();
            Integer quantity = entry.getValue();
            resultActions.andExpect( jsonPath( "$.positions.[?(@.merchandiseId == '" + merchandiseId + "')].quantity" )
                    .value( quantity ) );
            MoneyType salesPrice = merchandiseCatalogUseCases.getSalesPrice( merchandiseId );
            totalSalesPriceAsFloat += salesPrice.getAmount() * quantity;
        }
        // last thing - check the total sales price
        String totalSalesPriceString = String.format( "%.2f €", totalSalesPriceAsFloat );
        resultActions
                .andExpect( jsonPath( "$.totalSalesPrice" ).value( totalSalesPriceString ) );
    }

        if ( expectedCustomerName == null ) return;
        for ( PurchaseDTO purchase : purchases ) {
            if ( !purchase.getCustomerName().equals( expectedCustomerName ) ) {

    private void checkExpectedPurchases( Map<UUID, Map<UUID, Integer>> expectedPurchases,
                                            PurchaseDTO[] purchases ) {
        if ( expectedPurchases == null ) return;
        for ( PurchaseDTO purchase : purchases ) {
            UUID storageUnitId = purchase.getStorageUnitId();
            Map<UUID, Integer> expectedQuantities = expectedPurchases.get( storageUnitId );
            if ( expectedQuantities == null ) {
                throw new AssertionError( "No expected quantities for storage unit " + storageUnitId );
            }
            QuantityDTO[] positions = purchase.getPositions();
            for ( QuantityDTO quantityDTO : positions ) {
                UUID merchandiseId = quantityDTO.getMerchandiseId();
                Integer expectedQuantity = expectedQuantities.get( merchandiseId );
                if ( expectedQuantity == null ) {
                    throw new AssertionError( "No expected quantity for merchandise " + merchandiseId +
                            " in storage unit " + storageUnitId );
                }
                if ( !quantityDTO.getQuantity().equals( expectedQuantity ) ) {
                    throw new AssertionError( "Expected quantity " + expectedQuantity + " for merchandise " +
                            merchandiseId + " in storage unit " + storageUnitId +
                            ", but got " + quantityDTO.getQuantity() );
                }
            }
        }

import java.util.HashSet;
import java.util.Set;

import static org.junit.jupiter.api.Assertions.assertTrue;

public class BasicRequiredClassesTest {
    private static final String[] REQUIRED_CLASSES = {
            "Customer",
            "CustomerRepository",
            "Merchandise",
            "MerchandiseRepository",
            "Purchase",
            "PurchaseRepository",
            "InventoryLevel",
            "ShoppingCart",
            "ShoppingCartRepository",
            "StorageUnitRepository",
            "ShoppingCartPosition",
            "PurchasePosition",
            "StorageUnit",
    };

    @Test
    public void testRequiredClassesDeclared() throws IOException {
        String basePackage = "thkoeln.archilab.ecommerce";
        String packageSearchPath = "classpath*:" + basePackage.replace( '.', '/' ) + "/**/*.class";

        PathMatchingResourcePatternResolver resolver = new PathMatchingResourcePatternResolver();
        MetadataReaderFactory readerFactory = new CachingMetadataReaderFactory( resolver );

        Set<String> foundClasses = new HashSet<>();

        Arrays.stream( resolver.getResources( packageSearchPath ) ).forEach( resource -> {
            try {
                MetadataReader reader = readerFactory.getMetadataReader( resource );
                foundClasses.add( reader.getClassMetadata().getClassName() );
            } catch (IOException e) {
                e.printStackTrace();
            }
        } );

        Arrays.stream(REQUIRED_CLASSES).forEach(requiredClass -> {
            boolean found = foundClasses.stream().anyMatch(className ->
                    className.startsWith(basePackage) && className.endsWith("." + requiredClass)
            );
            assertTrue(found, "Expected class not found: " + basePackage + ".*." + requiredClass);
        });
    }
}

        put("CustomerRegistrationUseCases", Arrays.asList("customer"));

    private static final String SOLUTION_PATH = "thkoeln.archilab.ecommerce.solution";

    @Test
    public void testRequiredClassesDeclared() {
        for ( Map.Entry<String, List<String>> entry : USECASE_INTERFACES.entrySet() ) {
            String interfaceName = USECASE_PATH + "." + entry.getKey();
            List<String> correctTopLevelPackages = entry.getValue();
            Class<?> useCaseInterfaceClass = null;
            try {
                useCaseInterfaceClass = Class.forName( interfaceName );
            } catch (ClassNotFoundException e) {
                fail( "Use case interface " + interfaceName + " not found.", e );
                continue;
            }
            String[] beans = applicationContext.getBeanNamesForType( useCaseInterfaceClass );
            assertEquals( 1, beans.length,
                    "There should be exactly 1 service implementation for the interface " +
                            entry.getKey() + ", but found " + beans.length );
            Object bean = applicationContext.getBean( beans[0] );
            boolean isValid = false;
            for ( String correctTopLevelPackage : correctTopLevelPackages ) {
                String expectedPath = SOLUTION_PATH + "." + correctTopLevelPackage;
                if ( bean.getClass().getPackage().getName().startsWith( expectedPath ) ) {
                    isValid = true;
                    break;
                }
            }
            assertTrue( isValid,
                    "Service " + bean.getClass().getSimpleName() + " should be located somewhere below "
                            + SOLUTION_PATH + " and any of " + correctTopLevelPackages );
        }

    private static ArchCondition<JavaField> beCollectionOfEntityOrValueObjectFromSamePackageOrDomainPrimitive( final String allowedPackage ) {
        return new ArchCondition<JavaField>(
                "reference a List or Set of entities or value objects from same aggregate, or DPs" ) {
            @Override
            public void check( JavaField field, ConditionEvents events ) {
                JavaClass rawType = field.getRawType();
                String rawTypeName = rawType.getFullName();
                if ( !rawTypeName.equals( "java.util.List" ) && !rawTypeName.equals( "java.util.Set" ) ) {
                    events.add( SimpleConditionEvent.violated( field,
                            field.getFullName() + " is not a List or Set (actual: " + rawTypeName + ")" ) );
                    return;
                }
                JavaType fieldType = field.getType();
                if ( !( fieldType instanceof JavaParameterizedType ) ) {
                    events.add( SimpleConditionEvent.violated( field,
                            field.getFullName() + " is a raw List/Set without a type argument" ) );
                    return;
                }
                List<JavaType> typeArguments = ( (JavaParameterizedType) fieldType ).getActualTypeArguments();
                if ( typeArguments.isEmpty() ) {
                    events.add( SimpleConditionEvent.violated( field,
                            field.getFullName() + " has no type arguments" ) );
                    return;
                }
                JavaType elementType = typeArguments.get( 0 );
                if ( !( elementType instanceof JavaClass ) ) {
                    events.add( SimpleConditionEvent.violated( field,
                            field.getFullName() + " has an unknown element type" ) );
                    return;
                }
                JavaClass elementClass = (JavaClass) elementType;
                boolean hasAnnotation =
                        elementClass.isAnnotatedWith( Entity.class ) || elementClass.isAnnotatedWith( Embeddable.class );
                boolean inSamePackage = elementClass.getPackageName().startsWith( allowedPackage );
                boolean isGenericId = elementClass.isAssignableTo( GenericId.class );
                boolean isDomainPrimitive = elementClass.getPackageName().startsWith( "thkoeln.archilab.ecommerce.domainprimitives" );
                boolean ok = ( isGenericId || ( hasAnnotation && inSamePackage ) || isDomainPrimitive );
                if ( !ok ) {
                    events.add( SimpleConditionEvent.violated( field,
                            field.getFullName() + " refers to " + elementClass.getFullName()
                                    + " which is not a DP, or an entity/value object in " + allowedPackage ) );
                }
            }
        };
    }

    private void checkMerchandiseNumbers( Set<UUID> purchaseIds, UUID storageUnitId, Map<UUID, Integer> expectedMap ) {
        UUID fittingPurchaseId = null;
        for ( UUID purchaseId : purchaseIds ) {
            UUID foundStorageUnitId = purchaseUseCases.getStorageUnitIdForPurchase( purchaseId );
            if ( foundStorageUnitId.equals( storageUnitId ) ) {
                fittingPurchaseId = purchaseId;
                break;
            }
        }
        assertNotNull( fittingPurchaseId, "No fitting purchase found for storage unit " + storageUnitId );

        int totalExpected = 0;
        int totalActual = 0;
        for ( Map.Entry<UUID, Integer> entry : expectedMap.entrySet() ) {
            totalExpected += entry.getValue();
        }
        log.info( "Check purchase numbers for merchandise {} in storage unit {}", fittingPurchaseId, storageUnitId );
        for ( Map.Entry<UUID, Integer> entry : expectedMap.entrySet() ) {
            UUID merchandiseId = entry.getKey();
            Integer foundNumber =
                    purchaseUseCases.getPurchaseQuantOfMerchandise( fittingPurchaseId, merchandiseId );
            assertEquals( entry.getValue(), foundNumber );
            totalActual += foundNumber;
        }
        // last check - do the total sums match?
        assertEquals( totalExpected, totalActual );
    }


    private void checkStorageUnitIds( Set<UUID> purchaseIds, UUID... expectedStorageUnitIds ) {
        assertEquals( expectedStorageUnitIds.length, purchaseIds.size() );
        Set<UUID> foundStorageUnitIds = new HashSet<>();
        for ( UUID purchaseId : purchaseIds ) {
            UUID storageUnitId = purchaseUseCases.getStorageUnitIdForPurchase( purchaseId );
            foundStorageUnitIds.add( storageUnitId );
        }
        for ( UUID expectedStorageUnitId : expectedStorageUnitIds ) {
            assertTrue( foundStorageUnitIds.contains( expectedStorageUnitId ) );
        }
    }
